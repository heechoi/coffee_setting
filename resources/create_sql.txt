-- db 존재하면 삭제
drop database if exists product_project;

-- db 생성
create database product_project;

-- db 사용
use product_project;

-- 제품정보
CREATE TABLE if not exists product_project.code (
    codeid char(10) ,
    codename char(10),
    PRIMARY KEY (codeid)
);

-- 기록입력테이블
CREATE TABLE if not exists product_project.record (
    pdno char(10),
    pdname char(10),
    price int(11),
    amount int(11),
    margin int(11),
    PRIMARY KEY (pdno),
    FOREIGN KEY (pdno) REFERENCES code(codeid)
    on delete casecade
);

-- 판매기록
CREATE TABLE if not exists product_project.sale(
	salepd char(10),
	saleprice int(11),
	surtax int(11),
	supplyprice int(11),
	mgrate int(11),
	
	
);
